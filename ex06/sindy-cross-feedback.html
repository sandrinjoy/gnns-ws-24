# 1.1 Simulation
# Function to compute the scalar product Theta(z, dz) \cdot Xi
def pendulum_rhs(zt, dzt, coefficients, terms):
    # Evaluate all terms in Theta(z, dz)
    Theta = np.array([term(zt, dzt) for term in terms]).T
    return Theta @ coefficients

# Function to define the ODE step for scipy.integrate.odeint
def pendulum_ode_step(y, t, coefficients, terms):
    z, dz = y  # y[0] = z(t), y[1] = dz(t)
    dz2 = pendulum_rhs(z, dz, coefficients, terms)  # Compute dz/dt
    return [dz, dz2]

# Function to simulate the pendulum
def simulate_pendulum(z0, dz0, coefficients, terms, T, dt):
    t = np.arange(0, T, dt)
    y0 = [z0, dz0]
    solution = odeint(pendulum_ode_step, y0, t, args=(coefficients, terms))
    z, dz = solution[:, 0], solution[:, 1]
    dz2 = pendulum_rhs(z, dz, coefficients, terms)
    return t, z, dz, dz2

# Function to create training data for the pendulum
def create_pendulum_data(z0_min, z0_max, dz0_min, dz0_max, coefficients, terms, T, dt, N, embedding=None, rejection=True):
    z_list, dz_list, dz2_list = [], [], []
    
    for _ in range(N):
        while True:
            z0 = np.random.uniform(z0_min, z0_max)
            dz0 = np.random.uniform(dz0_min, dz0_max)
            
            # Rejection condition
            if rejection and dz0**2 / 2 - np.cos(z0) > 0.99:
                continue

            t, z, dz, dz2 = simulate_pendulum(z0, dz0, coefficients, terms, T, dt)
            z_list.append(z)
            dz_list.append(dz)
            dz2_list.append(dz2)
            break

    return np.concatenate(z_list), np.concatenate(dz_list), np.concatenate(dz2_list)

# Define the library of functions Theta
terms = [
    lambda z, dz: np.ones_like(z),         # Constant term
    lambda z, dz: z,                      # z
    lambda z, dz: dz,                     # dz
    lambda z, dz: np.sin(z),              # sin(z)
    lambda z, dz: z**2,                   # z^2
    lambda z, dz: z * dz,                 # z * dz
    lambda z, dz: dz * np.sin(z),         # dz * sin(z)
    lambda z, dz: dz**2,                  # dz^2
    lambda z, dz: np.sin(z)**2            # sin(z)^2
]

# 1.2 Implementation & Training
# Sklearn-based SINDy implementation
def train_sindy_sklearn(z, dz, dz2, terms, lambda_):
    Theta = np.array([term(z, dz) for term in terms]).T
    model = Lasso(alpha=lambda_, fit_intercept=False)
    model.fit(Theta, dz2)
    return model.coef_

# PyTorch-based SINDy implementation
class SINDy(nn.Module):
    def __init__(self, num_terms):
        super(SINDy, self).__init__()
        self.coefficients = nn.Parameter(torch.ones(num_terms))
        self.mask = nn.Parameter(torch.ones(num_terms, dtype=torch.bool), requires_grad=False)

    def forward(self, z, dz, terms):
        Theta = torch.stack([torch.tensor(term(z, dz), dtype=torch.float32) for term in terms], dim=1)
        return Theta @ (self.coefficients * self.mask)

def train_sindy_pytorch(z, dz, dz2, terms, lambda_, num_epochs=1000, lr=1e-3):
    z, dz, dz2 = map(torch.tensor, (z, dz, dz2))
    z, dz, dz2 = z.float(), dz.float(), dz2.float()
    
    model = SINDy(len(terms))
    optimizer = optim.Adam(model.parameters(), lr=lr)

    for epoch in range(num_epochs):
        optimizer.zero_grad()
        dz2_pred = model(z, dz, terms)
        loss = torch.mean((dz2 - dz2_pred) ** 2) + lambda_ * torch.sum(torch.abs(model.coefficients))
        loss.backward()
        optimizer.step()

    return model

# Train both models

coefficients_pytorch = model_pytorch.coefficients.detach().numpy()

# Compare coefficients
print("Sklearn coefficients:", coefficients_sklearn)

# 1.3 Thresholding (Sequential and Patient)
def threshold_coefficients(coefficients, history, method="sequential", patience=5):
    if method == "sequential":
        return np.abs(coefficients) > 1e-2
    elif method == "patient":
        recent_history = history[-patience:]
        return np.all(np.abs(recent_history) > 1e-2, axis=0)
    else:
        return np.ones_like(coefficients, dtype=bool)

import numpy as np

# Function for Cartesian embedding
def embed_cartesian(z, dz, ddz):
    # Compute Cartesian position
    x = np.stack([np.sin(z), -np.cos(z)], axis=-1)
    
    # Compute Cartesian velocity
    dx = np.stack([np.cos(z) * dz, np.sin(z) * dz], axis=-1)
    
    # Compute Cartesian acceleration
    ddx = np.stack([
        -np.sin(z) * dz**2 + np.cos(z) * ddz,
        np.cos(z) * dz**2 + np.sin(z) * ddz
    ], axis=-1)
    
    return x, dx, ddx

import torch
import torch.nn as nn
import torch.optim as optim

# Define the Autoencoder architecture
class Autoencoder(nn.Module):
    def __init__(self, input_dim, latent_dim, hidden_dims):
        super(Autoencoder, self).__init__()
        
        # Encoder
        encoder_layers = []
        prev_dim = input_dim
        for h_dim in hidden_dims:
            encoder_layers.append(nn.Linear(prev_dim, h_dim))
            encoder_layers.append(nn.Sigmoid())
            prev_dim = h_dim
        encoder_layers.append(nn.Linear(prev_dim, latent_dim))  # Map to latent space
        self.encoder = nn.Sequential(*encoder_layers)
        
        # Decoder
        decoder_layers = []
        prev_dim = latent_dim
        for h_dim in reversed(hidden_dims):
            decoder_layers.append(nn.Linear(prev_dim, h_dim))
            decoder_layers.append(nn.Sigmoid())
            prev_dim = h_dim
        decoder_layers.append(nn.Linear(prev_dim, input_dim))  # Reconstruct input
        self.decoder = nn.Sequential(*decoder_layers)

    def forward(self, x):
        z = self.encoder(x)  # Encode
        x_reconstructed = self.decoder(z)  # Decode
        return z, x_reconstructed

# Training function
def train_autoencoder(autoencoder, data, epochs, learning_rate):
    optimizer = optim.Adam(autoencoder.parameters(), lr=learning_rate)
    criterion = nn.MSELoss()  # Reconstruction loss
    losses = []
    
    for epoch in range(epochs):
        optimizer.zero_grad()
        z, x_reconstructed = autoencoder(data)
        loss = criterion(x_reconstructed, data)  # Compute loss
        loss.backward()
        optimizer.step()
        losses.append(loss.item())
        
        if epoch % 100 == 0:
            print(f"Epoch {epoch}, Loss: {loss.item():.6f}")
    
    return losses

# Example usage
input_dim = 2  # Cartesian x has 2 components
latent_dim = 1  # z is 1D
hidden_dims = [16, 8]  # Example hidden layer configuration

# Create synthetic data

# Convert to PyTorch tensors

# Instantiate and train the autoencoder

losses = train_autoencoder(autoencoder, x_tensor, epochs=1000, learning_rate=1e-3)

class SigmoidDerivatives(nn.Module):
    def forward(self, x, dx, ddx):
        sig = torch.sigmoid(x)
        sig_prime = sig * (1 - sig)
        sig_double_prime = sig_prime * (1 - 2 * sig)
        
        z = sig
        dz = sig_prime * dx
        ddz = sig_double_prime * dx**2 + sig_prime * ddx
        return z, dz, ddz

class LinearDerivatives(nn.Module):
    def __init__(self, in_features, out_features):
        super(LinearDerivatives, self).__init__()
        self.linear = nn.Linear(in_features, out_features, bias=False)

    def forward(self, x, dx, ddx):
        z = self.linear(x)
        dz = self.linear(dx)
        ddz = self.linear(ddx)
        return z, dz, ddz

class SINDyAutoencoder(nn.Module):
    def __init__(self, autoencoder, sindy_model):
        super(SINDyAutoencoder, self).__init__()
        self.autoencoder = autoencoder  # Autoencoder instance
        self.sindy = sindy_model        # SINDy instance

    def forward(self, x, dx, ddx):
        # Encode to latent space
        z, dz, ddz = self.autoencoder.encoder[0](x, dx, ddx)
        
        # Use SINDy to compute predicted second derivative in latent space
        z_hat_ddot = self.sindy(z, dz)
        
        # Decode z_hat_ddot back to Cartesian space
        x_hat, dx_hat, x_hat_ddot = self.autoencoder.decoder[0](z, dz, z_hat_ddot)
        
        return x_hat, z_hat_ddot, x_hat_ddot

def train_sindy_autoencoder(
    model, 
    data, 
    epochs, 
    learning_rate, 
    lambda_z_ddot, 
    lambda_x_ddot, 
    lambda_L1, 
    refinement_epoch
):
    optimizer = optim.Adam(model.parameters(), lr=learning_rate)
    criterion = nn.MSELoss()  # MSE loss for reconstruction and derivatives
    
    for epoch in range(epochs):
        optimizer.zero_grad()
        
        # Forward pass
        x_hat, z_hat_ddot, x_hat_ddot = model(
            data["x"], data["dx"], data["ddx"]
        )
        
        # Loss terms
        reconstruction_loss = criterion(x_hat, data["x"])
        latent_loss = lambda_z_ddot * criterion(z_hat_ddot, data["z_ddot"])
        cartesian_loss = lambda_x_ddot * criterion(x_hat_ddot, data["ddx"])
        
        # L1 regularization for SINDy coefficients
        l1_loss = lambda_L1 * torch.norm(model.sindy.coefficients, p=1)
        if epoch >= refinement_epoch:
            l1_loss = 0  # Turn off L1 regularization after refinement begins
        
        # Total loss
        loss = reconstruction_loss + latent_loss + cartesian_loss + l1_loss
        loss.backward()
        optimizer.step()
        
        if epoch % 100 == 0:
            print(f"Epoch {epoch}, Loss: {loss.item():.6f}")

def sequential_thresholding(sindy, threshold, interval, epoch):
    if epoch % interval == 0:
        with torch.no_grad():
            mask = torch.abs(sindy.coefficients) > threshold
            sindy.coefficients *= mask  # Zero out small coefficients

def compute_fvu(y, y_pred):
    numerator = torch.sum((y - y_pred) ** 2)
    denominator = torch.sum((y - torch.mean(y)) ** 2)
    return (numerator / denominator).item()

def resimulate_sindy_autoencoder(model, z0, dz0, t_span, dt):
    # Time steps
    num_steps = int(t_span / dt)
    z = [z0]
    dz = [dz0]
    for _ in range(num_steps - 1):
        # Compute next z and dz using the SINDy model
        z_next = z[-1] + dz[-1] * dt
        dz_next = dz[-1] + model.sindy(z[-1], dz[-1]) * dt
        z.append(z_next)
        dz.append(dz_next)
    
    z_sim = torch.stack(z)
    dz_sim = torch.stack(dz)

    # Decode simulated z back into Cartesian coordinates
    x_sim, dx_sim, _ = model.autoencoder.decoder(z_sim, dz_sim, None)
    return z_sim, x_sim

import numpy as np

def embed_grid(z, t, resolution, sigma):
    T = len(t)  # Number of time steps
    L = 1.0  # Pendulum length
    grid = np.linspace(-2, 2, resolution)
    X, Y = np.meshgrid(grid, grid)

    video = np.zeros((T, resolution, resolution))
    for i in range(T):
        x_tip = L * np.sin(z[i])
        y_tip = -L * np.cos(z[i])
        video[i] = np.exp(-((X - x_tip)**2 + (Y - y_tip)**2) / (2 * sigma**2))
    
    # Compute time derivatives using finite differences
    dt = t[1] - t[0]
    dx = (video[2:] - video[:-2]) / (2 * dt)
    ddx = (video[:-2] - 2 * video[1:-1] + video[2:]) / (dt**2)
    
    return video, dx, ddx

class VideoAutoencoder(nn.Module):
    def __init__(self):
        super(VideoAutoencoder, self).__init__()
        # Encoder: 128 -> 64 -> 32
        self.encoder = nn.Sequential(
            nn.Linear(128*128, 128),
            nn.Sigmoid(),
            nn.Linear(128, 64),
            nn.Sigmoid(),
            nn.Linear(64, 32),
        )
        # Decoder: 32 -> 64 -> 128
        self.decoder = nn.Sequential(
            nn.Linear(32, 64),
            nn.Sigmoid(),
            nn.Linear(64, 128),
            nn.Sigmoid(),
            nn.Linear(128, 128*128),
            nn.Sigmoid(),
        )

    def forward(self, x):
        z = self.encoder(x)
        x_hat = self.decoder(z)
        return x_hat, z

import torch
import torch.nn as nn

class VideoAutoencoder(nn.Module):
    def __init__(self, resolution):
        super(VideoAutoencoder, self).__init__()
        input_size = resolution * resolution
        # Encoder
        self.encoder = nn.Sequential(
            nn.Linear(input_size, 128),
            nn.Sigmoid(),
            nn.Linear(128, 64),
            nn.Sigmoid(),
            nn.Linear(64, 32),
        )
        # Decoder
        self.decoder = nn.Sequential(
            nn.Linear(32, 64),
            nn.Sigmoid(),
            nn.Linear(64, 128),
            nn.Sigmoid(),
            nn.Linear(128, input_size),
            nn.Sigmoid(),
        )

    def forward(self, x):
        z = self.encoder(x)
        x_hat = self.decoder(z)
        return x_hat, z

def train_video_autoencoder(model, video, epochs=5000, batch_size=32):
    video = torch.tensor(video, dtype=torch.float32)
    dataset = torch.utils.data.TensorDataset(video)
    dataloader = torch.utils.data.DataLoader(dataset, batch_size=batch_size, shuffle=True)

    optimizer = torch.optim.Adam(model.parameters(), lr=1e-3)
    criterion = nn.MSELoss()

    for epoch in range(epochs):
        for batch in dataloader:
            x = batch[0]
            x_hat, _ = model(x)

            # Compute MSE loss
            loss = criterion(x_hat, x)

            optimizer.zero_grad()
            loss.backward()
            optimizer.step()

        if epoch % 100 == 0:
            print(f"Epoch {epoch}/{epochs}, Loss: {loss.item()}")

def compute_latent_derivatives(model, video_flat, dt):
    video_flat = torch.tensor(video_flat, dtype=torch.float32)
    _, z = model(video_flat)  # Latent variables (T, latent_dim)

    z = z.detach().numpy()
    dz = (z[2:] - z[:-2]) / (2 * dt)  # First derivative
    ddz = (z[:-2] - 2 * z[1:-1] + z[2:]) / (dt ** 2)  # Second derivative

    return z, dz, ddz

def compute_fvu(y, y_hat):
    numerator = np.sum((y - y_hat) ** 2)
    denominator = np.sum((y - np.mean(y)) ** 2)
    return numerator / denominator
